backup db:

docker:
docker build -t postgresql:raw . - pravi docker image
docker run --rm -P --name pg_test postgresql - pokreni bazu
psql -h localhost -p 32769 -d docker -U docker --password - idi gde je instaliran postgres/bin nadji psql tu otvori
comm line, -p je onaj port sto pise kad uradis docker ps -a (pa pogledas tu koji je port od postgres)
